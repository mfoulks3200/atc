# ATC Daemon Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement `@atc/daemon` — a Fastify-based long-running process that orchestrates agents, manages git projects, and exposes REST + WebSocket APIs — plus the `@atc/adapter-claude-agent-sdk` default adapter package.

**Architecture:** The daemon is a new `packages/daemon` package in the monorepo. It wraps existing `@atc/tower`, `@atc/core`, and `@atc/checklist` packages behind HTTP/WebSocket APIs. State is persisted as JSON files under `~/.atc/profiles/<name>/`. Agent lifecycle is managed through a pluggable adapter interface, with the Claude Agent SDK adapter in a separate `packages/adapter-claude-agent-sdk` package.

**Tech Stack:** Fastify + @fastify/websocket, Node.js child_process, @anthropic-ai/sdk (Agent SDK), vitest

**Spec:** `docs/superpowers/specs/2026-03-26-daemon-design.md`

```typescript
/**
 * Default values for global config.
 * @see docs/superpowers/specs/2026-03-26-daemon-design.md Section 3.1
 */
export const GLOBAL_CONFIG_DEFAULTS: GlobalConfig = {
  defaultProfile: "default",
};

/**
 * Default values for profile config.
 * @see docs/superpowers/specs/2026-03-26-daemon-design.md Section 3.2
 */
export const PROFILE_CONFIG_DEFAULTS: ProfileConfig = {
  port: 7700,
  host: "127.0.0.1",
  logLevel: "info",
  autoRecover: false,
  wsHeartbeatInterval: 15,
  stateFlushInterval: 30,
  adapter: {
    type: "claude-agent-sdk",
    config: {},
  },
};

/** Validates that a value is a valid port number. */
export function isValidPort(value: unknown): value is number {
  return typeof value === "number" && Number.isInteger(value) && value > 0 && value <= 65535;
}

/** Validates that a value is a valid log level. */
export function isValidLogLevel(value: unknown): value is ProfileConfig["logLevel"] {
  return typeof value === "string" && ["debug", "info", "warn", "error"].includes(value);
}

/**
 * Validates partial profile config. Throws on invalid field types.
 * Does not require all fields — missing fields will be filled from defaults.
 */
export function validatePartialProfileConfig(raw: Record<string, unknown>): void {
  if ("port" in raw && !isValidPort(raw.port)) {
    throw new Error(`Invalid port: ${JSON.stringify(raw.port)}. Must be an integer 1–65535.`);
  }
  if ("host" in raw && typeof raw.host !== "string") {
    throw new Error(`Invalid host: ${JSON.stringify(raw.host)}. Must be a string.`);
  }
  if ("logLevel" in raw && !isValidLogLevel(raw.logLevel)) {
    throw new Error(
      `Invalid logLevel: ${JSON.stringify(raw.logLevel)}. Must be debug, info, warn, or error.`,
    );
  }
  if ("autoRecover" in raw && typeof raw.autoRecover !== "boolean") {
    throw new Error(`Invalid autoRecover: ${JSON.stringify(raw.autoRecover)}. Must be a boolean.`);
  }
  if ("wsHeartbeatInterval" in raw && typeof raw.wsHeartbeatInterval !== "number") {
    throw new Error(`Invalid wsHeartbeatInterval: must be a number.`);
  }
  if ("stateFlushInterval" in raw && typeof raw.stateFlushInterval !== "number") {
    throw new Error(`Invalid stateFlushInterval: must be a number.`);
  }
}
```

```typescript
import { readFile } from "node:fs/promises";
import { join } from "node:path";
import type { GlobalConfig, ProfileConfig, ProjectMetadata } from "../types.js";
import {
  GLOBAL_CONFIG_DEFAULTS,
  PROFILE_CONFIG_DEFAULTS,
  validatePartialProfileConfig,
} from "./schema.js";

/**
 * Reads and parses a JSON file. Returns null if the file does not exist.
 */
async function readJsonFile(filePath: string): Promise<unknown | null> {
  try {
    const content = await readFile(filePath, "utf-8");
    return JSON.parse(content);
  } catch (err: unknown) {
    if (err instanceof Error && "code" in err && (err as NodeJS.ErrnoException).code === "ENOENT") {
      return null;
    }
    throw err;
  }
}

/**
 * Loads global config from ~/.atc/config.json.
 * Returns defaults if the file does not exist.
 * @see docs/superpowers/specs/2026-03-26-daemon-design.md Section 3.1
 */
export async function loadGlobalConfig(atcDir: string): Promise<GlobalConfig> {
  const raw = await readJsonFile(join(atcDir, "config.json"));
  if (raw === null) {
    return { ...GLOBAL_CONFIG_DEFAULTS };
  }
  const obj = raw as Record<string, unknown>;
  return {
    defaultProfile:
      typeof obj.defaultProfile === "string"
        ? obj.defaultProfile
        : GLOBAL_CONFIG_DEFAULTS.defaultProfile,
  };
}

/**
 * Resolves the filesystem path for a profile.
 * @see docs/superpowers/specs/2026-03-26-daemon-design.md Section 2
 */
export function resolveProfilePath(atcDir: string, profileName?: string): string {
  return join(atcDir, "profiles", profileName ?? "default");
}

/**
 * Loads profile config, merging with defaults.
 * Returns all defaults if the file does not exist.
 * Throws on invalid field types.
 * @see docs/superpowers/specs/2026-03-26-daemon-design.md Section 3.2
 */
export async function loadProfileConfig(profileDir: string): Promise<ProfileConfig> {
```

```typescript
import { readFile, writeFile, rename, mkdir } from "node:fs/promises";
import { dirname, join } from "node:path";
import { randomBytes } from "node:crypto";

/**
 * Writes JSON to a file atomically using tmp+rename.
 * Creates parent directories if they do not exist.
 * @see docs/superpowers/specs/2026-03-26-daemon-design.md Section 4.5
 */
export async function atomicWriteJson(filePath: string, data: unknown): Promise<void> {
  const dir = dirname(filePath);
  await mkdir(dir, { recursive: true });
  const tmpPath = join(dir, `.${randomBytes(8).toString("hex")}.tmp`);
  const json = JSON.stringify(data, null, 2);
  await writeFile(tmpPath, json, "utf-8");
  await rename(tmpPath, filePath);
}

/**
 * Reads and parses a JSON file. Returns null if the file does not exist.
 */
export async function readJsonSafe<T = unknown>(filePath: string): Promise<T | null> {
  try {
    const content = await readFile(filePath, "utf-8");
    return JSON.parse(content) as T;
  } catch (err: unknown) {
    if (err instanceof Error && "code" in err && (err as NodeJS.ErrnoException).code === "ENOENT") {
      return null;
    }
    throw err;
  }
}

/**
 * Manages periodic state flushing with dirty tracking.
 * @see docs/superpowers/specs/2026-03-26-daemon-design.md Section 4.5
 */
export class FlushScheduler {
  private dirty = false;
  private timer: ReturnType<typeof setInterval> | null = null;
  private flushFn: () => Promise<void>;

  constructor(flushFn: () => Promise<void>, intervalSeconds?: number) {
    this.flushFn = flushFn;
    if (intervalSeconds && intervalSeconds > 0) {
      this.timer = setInterval(() => {
        if (this.dirty) {
          void this.flush();
        }
      }, intervalSeconds * 1000);
    }
  }

  get isDirty(): boolean {
    return this.dirty;
  }

  markDirty(): void {
    this.dirty = true;
  }

  async flush(): Promise<void> {
    await this.flushFn();
    this.dirty = false;
  }

  stop(): void {
    if (this.timer) {
      clearInterval(this.timer);
      this.timer = null;
    }
  }
}
```

```typescript
/**
 * In-memory agent registry with disk persistence.
 * @see docs/superpowers/specs/2026-03-26-daemon-design.md Section 4.5
 */
export class AgentStore {
  private agents = new Map<string, AgentRecord>();
  private stateDir: string;

  constructor(stateDir: string) {
    this.stateDir = stateDir;
  }

  get(id: string): AgentRecord | undefined {
    return this.agents.get(id);
  }

  set(record: AgentRecord): void {
    this.agents.set(record.id, record);
  }

  remove(id: string): void {
    this.agents.delete(id);
  }

  list(): AgentRecord[] {
    return [...this.agents.values()];
  }

  updateStatus(id: string, status: AgentStatus): void {
    const agent = this.agents.get(id);
    if (agent) {
      agent.status = status;
    }
  }

  async save(): Promise<void> {
    await atomicWriteJson(join(this.stateDir, "agents.json"), this.list());
  }

  async load(): Promise<void> {
    const data = await readJsonSafe<AgentRecord[]>(join(this.stateDir, "agents.json"));
    this.agents.clear();
    if (data) {
      for (const record of data) {
        this.agents.set(record.id, record);
      }
    }
  }
}
```

```typescript
    this.stateDir = stateDir;

  set(projectName: string, craft: CraftState): void {
    if (!this.crafts.has(projectName)) {
      this.crafts.set(projectName, new Map());

    this.crafts.get(projectName)!.set(craft.callsign, craft);

    return map ? [...map.values()] : [];

    const craft = this.get(projectName, callsign);
    if (craft) {
      craft.intercom.push(message);
    }

    return join(this.stateDir, "projects", projectName, "crafts", callsign);

    const craft = this.get(projectName, callsign);
    if (craft) {
      await atomicWriteJson(join(this.craftDir(projectName, callsign), "craft.json"), craft);
    }
```

```typescript
import { join } from "node:path";
import { atomicWriteJson, readJsonSafe } from "./persistence.js";

/** A single entry in the FCFS merge queue. */
export interface QueueEntry {
  callsign: string;
  requestedAt: string;
}

/**
 * Per-project tower merge queue state.
 * @see RULE-TMRG-4 — merges sequenced FCFS by default.
 */
export class TowerStore {
  private queues = new Map<string, QueueEntry[]>();
  private stateDir: string;

  constructor(stateDir: string) {
    this.stateDir = stateDir;
  }

  getQueue(projectName: string): QueueEntry[] {
    return this.queues.get(projectName) ?? [];
  }

  enqueue(projectName: string, callsign: string): void {
    if (!this.queues.has(projectName)) {
      this.queues.set(projectName, []);
    }
    this.queues.get(projectName)!.push({
      callsign,
      requestedAt: new Date().toISOString(),
    });
  }

  dequeue(projectName: string, callsign: string): void {
    const queue = this.queues.get(projectName);
    if (queue) {
      this.queues.set(
        projectName,
        queue.filter((e) => e.callsign !== callsign),
      );
    }
  }

  async save(projectName: string): Promise<void> {
    const filePath = join(this.stateDir, "projects", projectName, "tower.json");
    await atomicWriteJson(filePath, this.getQueue(projectName));
  }

  async load(projectName: string): Promise<void> {
    const filePath = join(this.stateDir, "projects", projectName, "tower.json");
    const data = await readJsonSafe<QueueEntry[]>(filePath);
    if (data) {
      this.queues.set(projectName, data);
    }
  }
}
```

```typescript
/**
 * Registry of agent adapters by name.
 * @see docs/superpowers/specs/2026-03-26-daemon-design.md Section 5.2
 */
export class AdapterRegistry {
  private adapters = new Map<string, AgentAdapter>();

  register(name: string, adapter: AgentAdapter): void {
    if (this.adapters.has(name)) {
      throw new Error(`Adapter "${name}" is already registered`);
    }
    this.adapters.set(name, adapter);
  }

  get(name: string): AgentAdapter | undefined {
    return this.adapters.get(name);
  }

  list(): string[] {
    return [...this.adapters.keys()];
  }
}
```

```typescript
import { exec } from "node:child_process";
import type { ChecklistItemConfig } from "../types.js";

const DEFAULT_TIMEOUT_SECONDS = 120;

/** Result for a single checklist item. */
export interface ChecklistItemResult {
  name: string;
  passed: boolean;
  stdout: string;
  stderr: string;
  error?: string;
  durationMs: number;
}

/** Aggregate result for the full checklist run. */
export interface ChecklistResult {
  passed: boolean;
  items: ChecklistItemResult[];
}

/**
 * Runs a sequence of checklist commands in the given working directory.
 * Stops on first failure (per RULE-LCHK-3).
 * @see docs/superpowers/specs/2026-03-26-daemon-design.md Section 3.3
 */
export async function runChecklist(
  items: ChecklistItemConfig[],
  cwd: string,
): Promise<ChecklistResult> {
  const results: ChecklistItemResult[] = [];

  for (const item of items) {
    const timeoutMs = (item.timeout ?? DEFAULT_TIMEOUT_SECONDS) * 1000;
    const start = Date.now();

    const result = await runCommand(item.command, cwd, timeoutMs);
    const durationMs = Date.now() - start;

    const itemResult: ChecklistItemResult = {
      name: item.name,
      passed: result.passed,
      stdout: result.stdout,
      stderr: result.stderr,
      error: result.error,
      durationMs,
    };

    results.push(itemResult);

    if (!result.passed) {
      return { passed: false, items: results };
    }
  }

  return { passed: true, items: results };
}

interface CommandResult {
  passed: boolean;
  stdout: string;
  stderr: string;
  error?: string;
}

function runCommand(command: string, cwd: string, timeoutMs: number): Promise<CommandResult> {
  return new Promise((resolve) => {
    const child = exec(command, { cwd, timeout: timeoutMs }, (err, stdout, stderr) => {
      if (err) {
        const isTimeout = err.killed || ("code" in err && err.code === "ERR_CHILD_PROCESS_STDIO_MAXBUFFER");
        resolve({
          passed: false,
          stdout: stdout ?? "",
          stderr: stderr ?? "",
          error: isTimeout ? `Command "${command}" timed out after ${timeoutMs / 1000}s` : undefined,
        });
      } else {
        resolve({ passed: true, stdout: stdout ?? "", stderr: stderr ?? "" });
      }
    });

    // Handle timeout edge case — exec's timeout sends SIGTERM
    child.on("error", () => {
      resolve({ passed: false, stdout: "", stderr: "", error: `Failed to execute: ${command}` });
    });
  });
}
```

```typescript
import { readFile, writeFile, rm } from "node:fs/promises";

/**
 * Writes the current process PID to a file.
 * @see docs/superpowers/specs/2026-03-26-daemon-design.md Section 4.1
 */
export async function writePidFile(pidPath: string): Promise<void> {
  await writeFile(pidPath, String(process.pid), "utf-8");
}

/**
 * Reads a PID from a file. Returns null if the file does not exist.
 */
export async function readPidFile(pidPath: string): Promise<number | null> {
  try {
    const content = await readFile(pidPath, "utf-8");
    const pid = parseInt(content.trim(), 10);
    return isNaN(pid) ? null : pid;
  } catch (err: unknown) {
    if (err instanceof Error && "code" in err && (err as NodeJS.ErrnoException).code === "ENOENT") {
      return null;
    }
    throw err;
  }
}

/**
 * Removes the PID file. Does not throw if it doesn't exist.
 * @see docs/superpowers/specs/2026-03-26-daemon-design.md Section 4.2
 */
export async function removePidFile(pidPath: string): Promise<void> {
  await rm(pidPath, { force: true });
}

/**
 * Checks if a process with the given PID is alive.
 */
export function isProcessAlive(pid: number): boolean {
  try {
    process.kill(pid, 0);
    return true;
  } catch {
    return false;
  }
}
```

```typescript
/**
 * Creates a shutdown handler that ensures the callback is only called once.
 * Designed to be attached to SIGTERM and SIGINT.
 * @see docs/superpowers/specs/2026-03-26-daemon-design.md Section 4.2
 */
export function createShutdownHandler(
  onShutdown: () => Promise<void>,
): () => Promise<void> {
  let shutdownInProgress = false;

  return async () => {
    if (shutdownInProgress) return;
    shutdownInProgress = true;
    await onShutdown();
  };
}

/**
 * Registers SIGTERM and SIGINT handlers that invoke the given shutdown function.
 */
export function registerSignalHandlers(handler: () => Promise<void>): void {
  process.on("SIGTERM", () => void handler());
  process.on("SIGINT", () => void handler());
}
```

```typescript
/**
 * Manages WebSocket channel subscriptions and event publishing.
 * Uses glob-style prefix matching for wildcard channels.
 * @see docs/superpowers/specs/2026-03-26-daemon-design.md Section 8.3
 */
export class ChannelRegistry {
  /** Map<clientId, Map<channelPattern, sendFn>> */
  private subscriptions = new Map<string, Map<string, (data: unknown) => void>>();

  subscribe(clientId: string, channel: string, send: (data: unknown) => void): void {
    if (!this.subscriptions.has(clientId)) {
      this.subscriptions.set(clientId, new Map());
    }
    this.subscriptions.get(clientId)!.set(channel, send);
  }

  unsubscribe(clientId: string, channel: string): void {
    this.subscriptions.get(clientId)?.delete(channel);
  }

  removeClient(clientId: string): void {
    this.subscriptions.delete(clientId);
  }

  getSubscriptions(clientId: string): string[] {
    const map = this.subscriptions.get(clientId);
    return map ? [...map.keys()] : [];
  }

  publish(channel: string, data: unknown): void {
    for (const [, clientSubs] of this.subscriptions) {
      for (const [pattern, send] of clientSubs) {
        if (channelMatches(pattern, channel)) {
          send(data);
        }
      }
    }
  }
}

/**
 * Checks if a subscription pattern matches a channel.
 * "*" matches everything. "foo:*" matches any channel starting with "foo:".
 */
function channelMatches(pattern: string, channel: string): boolean {
  if (pattern === "*") return true;
  if (pattern === channel) return true;
  if (pattern.endsWith(":*")) {
    const prefix = pattern.slice(0, -1); // "foo:*" → "foo:"
    return channel.startsWith(prefix);
  }
  return false;
}
```

```typescript
/**
 * Tracks WebSocket client liveness via ping/pong.
 * @see docs/superpowers/specs/2026-03-26-daemon-design.md Section 8.5
 */
export class HeartbeatTracker {
  private missedPongs = new Map<string, number>();
  private maxMissed: number;

  constructor(maxMissed: number) {
    this.maxMissed = maxMissed;
  }

  addClient(clientId: string): void {
    this.missedPongs.set(clientId, 0);
  }

  removeClient(clientId: string): void {
    this.missedPongs.delete(clientId);
  }

  /** Called each heartbeat interval — increments missed count for all clients. */
  tick(): void {
    for (const [clientId, count] of this.missedPongs) {
      this.missedPongs.set(clientId, count + 1);
    }
  }

  /** Called when a pong is received — resets the client's missed count. */
  receivePong(clientId: string): void {
    if (this.missedPongs.has(clientId)) {
      this.missedPongs.set(clientId, 0);
    }
  }

  getMissedPongs(clientId: string): number {
    return this.missedPongs.get(clientId) ?? 0;
  }

  /** Returns client IDs that have missed >= maxMissed pongs. */
  getStaleClients(): string[] {
    const stale: string[] = [];
    for (const [clientId, count] of this.missedPongs) {
      if (count >= this.maxMissed) {
        stale.push(clientId);
      }
    }
    return stale;
  }
}
```

```typescript
import { loadProjectMetadata } from "../../config/loader.js";
import { atomicWriteJson } from "../../state/persistence.js";

import type { ProjectMetadata } from "../../types.js";

/**
 * Registers project CRUD routes.
 * @see docs/superpowers/specs/2026-03-26-daemon-design.md Section 7.2
 */
export async function projectRoutes(app: FastifyInstance): Promise<void> {
  const projectsDir = () => join(app.profileDir, "projects");

  app.post<{ Body: Omit<ProjectMetadata, "mcpServers"> & { mcpServers?: Record<string, unknown> } }>(
    "/api/v1/projects",
    async (request, reply) => {
      const { name, remoteUrl, categories, checklist, mcpServers } = request.body;
      const projectDir = join(projectsDir(), name);
      const meta: ProjectMetadata = {
        name,
        remoteUrl,
        categories: categories ?? [],
        checklist: checklist ?? [],
        mcpServers: (mcpServers ?? {}) as ProjectMetadata["mcpServers"],
      };

      await mkdir(projectDir, { recursive: true });
      await atomicWriteJson(join(projectDir, "metadata.json"), meta);
      await mkdir(join(projectDir, "crafts"), { recursive: true });

      try {
        await cloneBareRepo(remoteUrl, join(projectDir, "repo.git"));
      } catch {
        // Bare repo clone may fail in tests with local paths; folder structure is still valid
      }

      void reply.code(201);
      return meta;
    },
  );

  app.get("/api/v1/projects", async () => {
    let entries: string[];
    try {
      entries = await readdir(projectsDir());
    } catch {
      return [];
    }
    const projects: ProjectMetadata[] = [];
    for (const name of entries) {
      try {
        const meta = await loadProjectMetadata(join(projectsDir(), name));
        projects.push(meta);
      } catch {
        // skip directories without valid metadata
      }
    }
    return projects;
  });

  app.get<{ Params: { name: string } }>("/api/v1/projects/:name", async (request, reply) => {
    try {
      return await loadProjectMetadata(join(projectsDir(), request.params.name));
    } catch {
      void reply.code(404);
      return { error: "Project not found" };
    }
  });

  app.delete<{ Params: { name: string } }>("/api/v1/projects/:name", async (request, reply) => {
    const projectDir = join(projectsDir(), request.params.name);
    await rm(projectDir, { recursive: true, force: true });
    void reply.code(204);
    return;
  });

  app.post<{ Params: { name: string } }>("/api/v1/projects/:name/sync", async (request, reply) => {
    const projectDir = join(projectsDir(), request.params.name);
    try {
      await loadProjectMetadata(projectDir);
    } catch {
      void reply.code(404);
      return { error: "Project not found" };
```

```typescript
import type { WsClientMessage, WsServerMessage } from "../../types.js";
import type { ChannelRegistry } from "./channels.js";
import type { HeartbeatTracker } from "./heartbeat.js";

/**
 * Processes a single incoming WebSocket message.
 * @see docs/superpowers/specs/2026-03-26-daemon-design.md Section 8
 */
export function handleWsMessage(
  message: WsClientMessage,
  clientId: string,
  send: (data: WsServerMessage) => void,
  channels: ChannelRegistry,
  heartbeat: HeartbeatTracker,
): void {
  switch (message.type) {
    case "subscribe":
      channels.subscribe(clientId, message.channel, send);
      break;
    case "unsubscribe":
      channels.unsubscribe(clientId, message.channel);
      break;
    case "pong":
      heartbeat.receivePong(clientId);
      break;
    case "ping":
      send({ type: "pong", timestamp: new Date().toISOString() });
      break;
  }
}
```

```typescript
import { join } from "node:path";
import { createApp } from "./server/app.js";
import { loadProfileConfig } from "./config/loader.js";
import { AgentStore } from "./state/agent-store.js";
import { CraftStore } from "./state/craft-store.js";
import { TowerStore } from "./state/tower-store.js";
import { FlushScheduler } from "./state/persistence.js";
import { writePidFile, removePidFile } from "./process/pid.js";
import { createShutdownHandler, registerSignalHandlers } from "./process/signals.js";
import type { FastifyInstance } from "fastify";
import type { ProfileConfig } from "./types.js";

/**
 * Top-level daemon class that orchestrates all subsystems.
 * @see docs/superpowers/specs/2026-03-26-daemon-design.md Section 4
 */
export class Daemon {
  private profileDir: string;
  private app: FastifyInstance | null = null;
  private config: ProfileConfig | null = null;
  private agentStore: AgentStore;
  private craftStore: CraftStore;
  private towerStore: TowerStore;
  private flushScheduler: FlushScheduler | null = null;
  private running = false;
  private boundPort = 0;

  constructor(profileDir: string) {
    this.profileDir = profileDir;
    const stateDir = join(profileDir, "state");
    this.agentStore = new AgentStore(stateDir);
    this.craftStore = new CraftStore(stateDir);
    this.towerStore = new TowerStore(stateDir);
  }

  get isRunning(): boolean {
    return this.running;
  }

  get port(): number {
    return this.boundPort;
  }

  async start(): Promise<void> {
    this.config = await loadProfileConfig(this.profileDir);

    // Load persisted state
    await this.agentStore.load();

    // Create and start the HTTP/WS server
    this.app = createApp({ profileDir: this.profileDir });
    await this.app.listen({ port: this.config.port, host: this.config.host });

    const address = this.app.server.address();
    this.boundPort = typeof address === "object" && address ? address.port : this.config.port;

    // Start periodic state flush
    this.flushScheduler = new FlushScheduler(async () => {
      await this.agentStore.save();
      await this.craftStore.saveAll();
    }, this.config.stateFlushInterval);

    // Write PID file
    await writePidFile(join(this.profileDir, "daemon.pid"));

    // Register signal handlers
    const shutdown = createShutdownHandler(() => this.stop());
    registerSignalHandlers(shutdown);

    this.running = true;
  }

  async stop(): Promise<void> {
    if (!this.running) return;

    // Flush all state
    if (this.flushScheduler) {
      await this.flushScheduler.flush();
      this.flushScheduler.stop();
    }

    // Close server
    if (this.app) {
      await this.app.close();
    }

    // Remove PID file
    await removePidFile(join(this.profileDir, "daemon.pid"));

    this.running = false;
  }
}
```
